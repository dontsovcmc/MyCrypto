    import crypto as ut

    data = ut.load_data('5.txt').strip()
    result = ut.load_data('5r.txt')

    kL = 'ICE' * (len(data)/3 + 1)
    kL = [ord(c) for c in kL]

    rL = list()

    # loop through
    for c,k in zip(data,kL):
        h = hex(ord(c))[2:]
        ct = ut.xor_hex_chars(h,hex(k))
        ct = hex(ct)[2:]
        if len(ct) == 1:  ct = '0' + ct
        rL.append(ct)

    s2 = ''.join(rL)
    print s2 == result
